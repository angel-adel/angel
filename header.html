// menu.js — полностью рабочий бургер-меню document.addEventListener('DOMContentLoaded', function() { initMobileMenu(); }); // Функция инициализации (можно вызывать несколько раз — защита от дублирования) function initMobileMenu() { const toggleBtn = document.getElementById('menuToggle'); const nav = document.querySelector('nav'); if (!toggleBtn || !nav) return; // Защита от повторной инициализации if (toggleBtn.hasListener) return; toggleBtn.hasListener = true; // Обработчик клика по бургеру toggleBtn.addEventListener('click', function(e) { e.preventDefault(); e.stopPropagation(); nav.classList.toggle('open'); // Меняем иконку: ☰ → ✕ if (nav.classList.contains('open')) { toggleBtn.textContent = '✕'; toggleBtn.style.fontSize = '28px'; } else { toggleBtn.textContent = '☰'; } }); // Закрываем меню при клике на любую ссылку const links = nav.querySelectorAll('a'); links.forEach(link => { link.addEventListener('click', function() { nav.classList.remove('open'); toggleBtn.textContent = '☰'; }); }); } // Если шапка подгружается динамически (через fetch или innerHTML), // вызываем инициализацию ещё раз через небольшую задержку setTimeout(initMobileMenu, 150); setTimeout(initMobileMenu, 500);
